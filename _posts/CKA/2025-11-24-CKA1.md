```bash
kubectl run --help
# kubectl run 명령어 사용법과 옵션을 보는 도움말

kubectl get pods
# 현재 실행 중인 파드 목록을 확인

kubectl describe pod 파드이름
# 특정 파드의 상세 상태와 오류 원인을 확인

kubectl get pods -o wide
# 파드 목록을 노드 정보까지 포함해서 확인

kubectl delete pod 파드이름
# 해당 파드를 삭제

kubectl run nginx(=파드이름) --image=nginx(=이미지이름) --dry-run=client -o yaml
# 파드를 실제로 만들지 않고 YAML 파일 형태로 미리 출력

cat 파드이름.yaml
# YAML 파일 내용을 화면에 출력

vi 파드이름.yaml
# YAML 파일을 열어서 수정

kubectl apply -f 파드이름.yaml
# YAML 파일 내용대로 파드를 생성하거나 수정

```
